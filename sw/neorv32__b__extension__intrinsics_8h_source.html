68 uint32_t __attribute__ ((noinline)) riscv_intrinsic_clz(uint32_t rs1) {
69 
70  register uint32_t result __asm__ ("a0");
71  register uint32_t tmp_a __asm__ ("a0") = rs1;
72 
73  // dummy instruction to prevent GCC "constprop" optimization
74  asm volatile ("add x0, %[input_i], x0" : : [input_i] "r" (tmp_a));
75 
76  // clz a0, a0
77  CUSTOM_INSTR_R1_TYPE(0b0110000, 0b00000, a0, 0b001, a0, 0b0010011);
78 
79  return result;
80 }
